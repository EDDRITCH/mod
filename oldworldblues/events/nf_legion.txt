	option = {
		name = nf_legion.1.option.a

			base = 1
			modifier = {
				is_historical_focus_on = yes
				factor = 100
			}
		}
		
        set_country_flag = accepted_petition
		
		WHT = {
		    set_politics = {
			    ruling_party = elites 
				elections_allowed = no 
			}
			add_popularity = { ideology = elites popularity = 0.5 }
		}	
		
		add_opinion_modifier = { 
            target = CES 
            modifier = loyal_tribe
        }   
        reverse_add_opinion_modifier = { 
            target = CES
            modifier = loyal_tribe
        }
		
		declare_war_on = {
			target = NCN
			type = annex_everything
		}
		
	}

		name = nf_legion.1.option.b

			base = 1
			modifier = {
				is_historical_focus_on = no

			}
		}

# Caesar's Illness
# triggers civil war:
# Caesar: id = 1000
# Vulpes Inclulta (commander) id = 1001
# Vulpes Inculta (country leader) id = 1005
# Lucius (commander) id = 1002
# Lucius (country leader) id = 1006
# Lanius (commander) id = 1004
# Lanius (country leader) id = 1007
# Malpais (commander) id = 1003
country_event = {
	id = nf_legion.5
	title = nf_legion.5.title
	desc = nf_legion.5.desc
	picture = GFX_event_CES_vulpes

    is_triggered_only = no
    fire_only_once = yes
	trigger = {
		tag = CES
		has_country_flag = {
			flag = conquered_86
			days > 100
		}
		OR = {
			has_completed_focus = ces_the_burned_man
			has_completed_focus = ces_triumph_at_the_dam
		}
		NOT = {
			any_neighbor_country = {
				AND = {
					has_war_with = CES
					NOT = { original_tag = NCR }
					strength_ratio = { tag = CES ratio > 0.15 }
				}
			}
		}
		NOT = { surrender_progress > 15 } 
	}
	mean_time_to_happen = {
		days = 600
		modifier = {
			factor = 3
			is_ai = yes
		}
		modifier = {
			factor = 0.5
			NCR = {
				OR = {
					exists = no
					has_civil_war = yes
					has_capitulated = yes
				}
			}
		}
		modifier = {
			factor = 1.5
			is_historical_focus_on = yes
		}
		modifier = {
			factor = 1.5
			has_war_with = NCR
		}
	}
	
	
	# stay as Lanius
	option = {
		name = nf_legion.5.option.a
        
        kill_country_leader = yes
		
		create_country_leader  = {
			name = "Lanius"
			desc = "Lanius_DESC"
			picture = "Portrait_Lanius.dds"
			expire = "3000.1.1"
			ideology = feudalism
			id = 1007
			
			traits = {
				cruel_tyrant
				warmonger
			}
		}
		create_country_leader  = {
			name = "Vulpes Inculta"
			desc = "Inculta_DESC"
			picture = "Portrait_Inculta.dds"
			expire = "3000.1.1"
			ideology = oligarchy
			id = 1005
				
			traits = {
				cruel_tyrant
			}
		}
		if = {
			limit = {
				is_ai = no
			}
			start_civil_war = {
				ideology = ruler
				size = 0.55
				capital = 522
				states_filter = {
					owner = { original_tag = CES }
					is_CES_civil_war_non_eligable = no					
				}
				keep_unit_leaders = { 1001 }
			}
		}
		if = {
			limit = {
				is_ai = yes
			}
			start_civil_war = {
				ideology = ruler
				size = 0.25
				capital = 522
				states_filter = {
					owner = { original_tag = CES }
					is_CES_civil_war_non_eligable = no					
				}
				keep_unit_leaders = { 1003 1002 1004 }
			}
		}
		hidden_effect = {
			remove_unit_leader = 1001
			set_country_flag = legion_had_civil_war
		}
		random_country = {
			limit = {
				original_tag = CES
				has_government = ruler
			}
			set_cosmetic_tag = VLP
			set_country_flag = legion_had_civil_war
		}
		custom_effect_tooltip = lanius_chosen
	}
	# switch to vulpes_inculta
	option = {
		name = nf_legion.5.option.b
        hidden_effect = {
			every_owned_state = {
				set_state_flag = do_not_decimate
			}
		}
        kill_country_leader = yes
        		
		create_country_leader  = {
			name = "Vulpes Inculta"
			desc = "Inculta_DESC"
			picture = "Portrait_Inculta.dds"
			expire = "3000.1.1"
			ideology = oligarchy
			id = 1005
			
			traits = {
				cruel_tyrant
				vulpes_inculta
			}
		}
		
		set_politics = {
			ruling_party = ruler
			elections_allowed = no
		}
		
		hidden_effect = {
			359 = { set_state_name = "Fortification Hill" }
		}
				
		create_country_leader  = {
			name = "Lanius"
			desc = "Lanius_DESC"
			picture = "Portrait_Lanius.dds"
			expire = "3000.1.1"
			ideology = tribalism
			id = 1007
				
			traits = {
				cruel_tyrant
				warmonger
			}
		}
		if = {
			limit = {
				is_ai = no
			}
			start_civil_war = {
				ideology = elites
				size = 0.45
				capital = 522
				states_filter = {
					owner = { original_tag = CES }
					is_CES_civil_war_non_eligable = no					
				}
				keep_unit_leaders = { 1002 1003 1004 }
			}
		}
		if = {
			limit = {
				is_ai = yes
			}
			start_civil_war = {
				ideology = elites
				size = 0.70
				capital = 522
				states_filter = {
					owner = { original_tag = CES }
					is_CES_civil_war_non_eligable = no					
				}
				keep_unit_leaders = { 1002 1003 1004 }
			}
		}
		hidden_effect = {
			remove_unit_leader = 1001
			set_country_flag = legion_had_civil_war
		}
		random_country = {
			limit = {
				original_tag = CES
				has_government = elites
			}
			set_country_flag = legion_had_civil_war
			set_cosmetic_tag = CES
		}
		hidden_effect = {
			every_owned_state = {
				clr_state_flag = do_not_decimate
			}
		}
		
		set_cosmetic_tag = VLP
		
		custom_effect_tooltip = vulpes_chosen
	}
    
    option = {
		name = nf_legion.5.option.c
        
        trigger = {
            has_completed_focus = ces_goddess_diana
			has_completed_focus = ces_the_twin_temples
        }
        
		set_country_flag = caesar_survived
		
        add_stability = -0.1
		add_war_support = 0.05
		country_event = { id = nf_legion.6 }
        
	}
}
