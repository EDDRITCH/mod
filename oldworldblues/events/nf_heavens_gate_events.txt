country_event = { # Religious fanatics launch a coup in our puppet
	id = nf_event_hea.9
	title = nf_event_hea.9.t
	desc = nf_event_hea.9.d
	picture = GFX_event_HEA_bishop1

	is_triggered_only = yes

	option = { # Let them go
		name = nf_event_hea.9.a
		ai_chance = {
			base = 1
		}
		end_puppet = FROM
		HEA = { puppet = FROM }
	}
	option = { # Launch a counter-coup
		name = nf_event_hea.9.b
		ai_chance = {
			base = 7
		}
		custom_effect_tooltip = HEA_convertion_cw
		hidden_effect = {
			FROM = {
				if = {
					limit = {
						has_government = elites
					}
					start_civil_war = {
						ruling_party = ruler
						ideology = ruler
						size = 0.6
					}
				}
				else = {
					start_civil_war = {
						ruling_party = elites
						ideology = elites
						size = 0.6
					}
				}
			}
			every_country = { # We puppet the revolter
				limit = {
					original_tag = FROM
					NOT = { is_puppet_of = ROOT }
				}
				HEA = { puppet = PREV }
				if = {
					limit = { NOT = { has_war_with = FROM } }
					FROM = { declare_war_on = { target = PREV type = civil_war } }
				}
				PREV = {
					add_to_war = {
						targeted_alliance = FROM
						enemy = PREV
					}
				}
				HEA = {
					add_to_war = {
						targeted_alliance = PREV
						enemy = FROM
					}
				}
			}
		}
		HEA = { country_event = { id = nf_event_hea.90 } }
	}
}
