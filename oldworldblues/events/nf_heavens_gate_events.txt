	picture = GFX_event_NCR_settlers

    is_triggered_only = yes
    
	option = {
		name = nf_event_hea.6.a
		trigger = {
			MUR = { is_ai = yes } #Ai will always say yes
		}
		
		IF = { limit = { is_in_faction = no }
			create_faction = "defensive_league"
		} 
		
		add_to_faction = MUR
        add_ai_strategy = {
			type = alliance
			id = "MUR"
			value = 200
		}
		MUR = {
			add_ai_strategy = {
				type = alliance
				id = "HEA"
				value = 200
			}
		}
	}
	
	option = {
		name = nf_event_hea.6.b
		trigger = {
			MUR = { is_ai = no } #Player gets to say no
		}
	
		MUR = {
			country_event = { id = nf_event_hea.60 }
		}
	}
}
