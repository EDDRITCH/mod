country_event = { # A puppet requests NCR military deployment
	id = raid_trouble.4
	title = raid_trouble.4.t
	desc = raid_trouble.4
	#picture = 

	is_triggered_only = yes

	option = { # Deploy our military
		name = raid_trouble.4.a
		ai_chance = {
			base = 9
		}
		add_manpower = -400
		remove_400_weapons = yes
		effect_tooltip = {
			FROM = {
				add_autonomy_score = {
				    value = -200
				    localization = called_in_ncr_military
				}
			}
		}
		FROM = { country_event = { id = raid_trouble.5 days = 5 } } # NCR helps!
	}

	option = { # We don't have the resources to spare
		name = raid_trouble.4.b
		ai_chance = {
			base = 1
		}
		reverse_add_opinion_modifier = {
			target = FROM
			modifier = no_NCR_mil_help
		}
		FROM = { country_event = { id = raid_trouble.6 days = 5 } } # NCR refuses to help!
	}
}

	option = { # Deploy our military
		trigger = { # Backup to ensure the NCR has 1 or more ranger deployment
			check_variable = { ranger_deployments > 0 }
		}
		name = raid_trouble.7.a
		#ai_chance = {
		#	base = 9
		#}
		subtract_from_variable = { ranger_deployments = 1 }
		custom_effect_tooltip = lose_1_deployment

		remove_200_weapons = yes
		effect_tooltip = {
			FROM = {

				    value = -400
