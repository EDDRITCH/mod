add_namespace = faction

country_event = {
	id = faction.1 #Do The Nevada Pact
	title = faction.1.t
	desc = faction.1.d
	picture = GFX_report_event_fascist_speech

	is_triggered_only = yes

	option = {
		name = faction.1.a

		ai_chance = {
			factor = 1
			modifier = {
				is_historical_focus_on = yes
				factor = 100
			}
		}

		trigger = {
			NOT = {
				has_global_flag = nevadapact
			}
		}

		if = {
			limit = {
				tag = NEW
			}
			VLT = {
				country_event = faction.2
			}
		}
		if = {
			limit = {
				tag = VLT
			}
			NEW = {
				country_event = faction.2
			}
		}

	}

	option = {
		name = faction.1.b

		trigger = {
			has_global_flag = nevadapact
		}

		ai_chance = {
			factor = 1
			modifier = {
				is_historical_focus_on = yes
				factor = 100
			}
		}
		if = {
			limit = {
				VLT = { is_faction_leader = yes }
			}
			VLT = {
				add_to_faction = NEW
				add_ai_strategy = {
					type = alliance
					id = "NEW"
					value = 200
				}
			}
			NEW = {
				add_ai_strategy = {
					type = alliance
					id = "VLT"
					value = 200
				}
			}
		}
		if = {
			limit = {
				NEW = { is_faction_leader = yes }
			}
			NEW = {
				add_to_faction = VLT
				add_ai_strategy = {
					type = alliance
					id = "VLT"
					value = 200
				}
			}
			VLT = {
				add_ai_strategy = {
					type = alliance
					id = "NEW"
					value = 200
				}
			}
		}
	}
}
