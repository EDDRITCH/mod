	id = startup.1
	title = startup.1.title
	desc = startup.1.desc
	#hidden = yes
	is_triggered_only = yes

		NOT = { has_country_flag = setup_country_ideas }
	}

	}
	
	option = {
		name = startup.1.option.a 
	}
}

# Legion startup
country_event = {
	id = startup.2
	title = startup.2.title
	desc = startup.2.desc
	picture = GFX_event_CES_startup
	is_triggered_only = yes
	
	option = {
		name = startup.2.option.a
		NCR = { add_ideas = enhanced_legion_campaign }
		every_country = {
			limit = {
				AND = {
					is_legion_conquest = yes
					is_ai = yes
				}
			}
			add_ideas = enhanced_legion_campaign
		}

		ai_chance = {
			factor = 0
		}

			factor = 0
		}

					is_legion_conquest = yes

				}

		}
	}

		name = startup.2.option.c

	}

# Country Regional Flags for Factions
country_event = {
	id = startup.4

	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = {
				is_in_north_california = yes
			}
			set_country_flag = north_california_nation
		}
		if = {
			limit = {
				is_in_south_california = yes
			}
			set_country_flag = south_california_nation
		}
		if = {
			limit = {
				is_in_new_vegas = yes
			}
			set_country_flag = new_vegas_nation
		}
		if = {
			limit = {
				is_in_utah = yes
			}
			set_country_flag = utah_nation
		}
		if = {
			limit = {
				is_in_nevada = yes
			}
			set_country_flag = nevada_nation
		}
		if = {
			limit = {
				is_in_arizona_sonora = yes
			}
			set_country_flag = arizona_sonora_nation
		}
	}

	option = {
		name = startup.4.a
	}
}
