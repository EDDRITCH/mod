NCR_invade_CES = {
	enable = {
		tag = NCR

		OR = {
			has_war_with = CES
			is_justifying_wargoal_against = CES
			has_wargoal_against = CES
			CES = {
				is_justifying_wargoal_against = NCR
				has_wargoal_against = NCR
			}
		}
	}
	abort = {
		OR = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			is_in_faction_with = CES
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			surrender_progress > 0.5
		}
	}

	ai_strategy = { type = area_priority        id = arizona_sonora 			  value = 2000 }

	ai_strategy = { type = invade               id = "CES" 					  value = 5000 }
}

NCR_prepare_for_war = {
	enable = {
		tag = NCR
		has_government = democratic
		is_historical_focus_on = yes
		NOT = { 
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
		}
		NOT = { CES = { has_war_with = NCR } }
		NOT = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			is_in_faction_with = CES
		}
		
		NOT = { surrender_progress > 0 }
	}
	abort = {
		OR = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			is_in_faction_with = CES

			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			
			surrender_progress > 0
		}
	}
	
	ai_strategy = {
		type = conquer

		value = 100
	}

		type = antagonize
