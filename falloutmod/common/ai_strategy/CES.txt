CES_invade_NCR = {
	enable = {
		tag = CES

		OR = {
			has_war_with = NCR
			is_justifying_wargoal_against = NCR
			has_wargoal_against = NCR
			NCR = { 
				is_justifying_wargoal_against = CES
				has_wargoal_against = CES
			}
		}
	}
	abort = {
		OR = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			is_in_faction_with = CES
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			surrender_progress > 0.5
		}
	}

	ai_strategy = { type = area_priority        id = south_california 			  value = 2000 }
	ai_strategy = { type = invade               id = "NCR" 					  value = 5000 }
}

CES_prepare_for_war = {
	enable = {
		tag = CES
		NOT = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
		}
		NCR = { has_war_with = CES }
		NOT = { 
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			is_in_faction_with = NCR
			has_war_with = NCR
		}
		
		NOT = { surrender_progress > 0 }
	}
	abort = {
		OR = { 
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			is_in_faction_with = NCR
			has_war_with = NCR

			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			
			surrender_progress > 0
		}
	}
	
	ai_strategy = {
		type = conquer
		id = "NCR"
		value = 100
	}
	ai_strategy = {
		type = antagonize
		id = "NCR"
		value = 500
	}
}
