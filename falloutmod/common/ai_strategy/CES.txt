CES_invade_NCR = {
	enable = {
		tag = CES

		OR = {
			has_war_with = NCR
			is_justifying_wargoal_against = NCR
			has_wargoal_against = NCR
			NCR = { 
				is_justifying_wargoal_against = CES
				has_wargoal_against = CES
			}
		}
	}
	abort = {
		OR = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			is_in_faction_with = CES
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			surrender_progress > 0.5
		}
	}

	ai_strategy = { type = area_priority        id = south_california 			  value = 2000 }
	ai_strategy = { type = invade               id = "NCR" 					  value = 5000 }
}

CES_prepare_for_war = {
	enable = {
		tag = CES
		NOT = { 
			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
		}
		NCR = { has_war_with = CES }
		NOT = { 
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			is_in_faction_with = NCR
			has_war_with = NCR
		}
		
		NOT = { surrender_progress > 0 }
	}
	abort = {
		OR = { 
			NCR = { has_capitulated = yes }
			NCR = { exists = no }
			NCR = { is_puppet = yes }
			is_in_faction_with = NCR
			has_war_with = NCR

			CES = { has_capitulated = yes }
			CES = { exists = no }
			CES = { is_puppet = yes }
			
			surrender_progress > 0
		}
	}
	
	ai_strategy = {
		type = conquer
		id = "NCR"
		value = 100
	}
	ai_strategy = {
		type = antagonize
		id = "NCR"
		value = 500
	}
}
CES_focus_on_invading_hangdogs = {
	enable = {
		original_tag = CES
		OR = {
			has_war_with = HAN
			is_justifying_wargoal_against = HAN
			has_wargoal_against = HAN
		}
		HAN = { NOT = { is_in_faction_with = CES } }
		NOT = { has_war_with = NCR }
		NCR = {
			NOT = { is_justifying_wargoal_against = CES }
			NOT = { has_wargoal_against = CES }
		}
		NOT = { surrender_progress > 0.1 }
		NOT = {
			any_enemy_country = {
				AND = {
					NOT = {
						tag = HAN
					}
					strength_ratio	= { tag = CES ratio > 0.15 }
				}
			}
		}
		
	}
	abort = {
		OR = {
			HAN = {
				exists = no
				has_capitulated = yes
				is_in_faction_with = CES
			}
			AND = {
				NOT = { has_war_with = HAN }
				NOT = { is_justifying_wargoal_against = HAN }
				NOT = { has_wargoal_against = HAN }
			}
			any_enemy_country = {
				AND = {
					NOT = {
						tag = HAN
					}
					strength_ratio	= { tag = CES ratio > 0.15 }
				}
			}
			surrender_progress > 0.1
			has_war_with = NCR
			NCR = {
				is_justifying_wargoal_against = CES
				has_wargoal_against = CES
			}
		}
	}
	
	ai_strategy = {
		type = conquer
		id = "HAN"
		value = 500
	}
	ai_strategy = {
		type = antagonize
		id = "HAN"
		value = 500
	}
	ai_strategy = {
		type = invade
		id = "HAN"
		value = 500
	}
	ai_strategy = {
		type = area_priority
		id = colorado_newmexico
		value = 160
	}
}
