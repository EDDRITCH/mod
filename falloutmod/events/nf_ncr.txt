country_event = { #enclave civil war annexed, civil war starts
	id = nf_ncr.29
	title = nf_ncr.29.t
	desc = nf_ncr.29.d
	picture = GFX_event_NCR_battle
	hidden = no
	is_triggered_only = yes
	immediate = {
		# Allgood
		every_country = {
			limit = {
				original_tag = NCR
				has_government = intellectuals
			}
			ENC = { annex_country = { target = PREV transfer_troops = yes } }
		}
	}
	option = {
		name = nf_ncr.29.a
		custom_effect_tooltip = ncr_civil_war_starts
		hidden_effect = {
			ENC = {
				transfer_state = 63
				set_capital = 63
			}
		
			every_country = {
				limit = {
					OR = {	
						tag = BAR
						tag = BON
						tag = ENC
						tag = NCR
					}
				}
				every_country = {
					limit = {
						NOT = { tag = PREV }
						OR = {
							tag = BAR
							tag = BON
							tag = ENC
							tag = NCR
						}
					}
					declare_war_on = {
						target = PREV
						type = civil_war
					}
				}
			}
			NCR = {
				hidden_effect = {
					every_owned_state = {
						clr_state_flag = do_not_decimate
					}
				}
			}
			BON = {
				transfer_ncr_ideas_civil_war = yes
				hidden_effect = {
					
					every_owned_state = {
						clr_state_flag = do_not_decimate
					}
				}
			}
			ENC = {
				setup_NCR_enclave = yes
				hidden_effect = {
					every_owned_state = {
						clr_state_flag = do_not_decimate
					}
				}
			}
			BAR = {
				transfer_ncr_ideas_civil_war = yes
				hidden_effect = {
					every_owned_state = {
						clr_state_flag = do_not_decimate
					}
				}
			}
			if = {
				limit = {
					has_country_flag = ncr_switch_to_enclave
					is_ai = no
				}
				ENC = { change_tag_from	= NCR }
			}
		}
	}
}
